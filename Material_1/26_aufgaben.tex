\begin{exercisebox}[Zentrale Differenz und zweite Differenz]
	\begin{itemize}
		\item[(a)] Zeigen Sie, dass sich die erste Ableitung von Polynomen vom Grad $2$ mit der zentralen Differenz exakt berechnen läßt.
		\item[(b)] Zeigen Sie, dass sich die zweite Ableitung von Polynomen vom Grad $4$ mit der zweiten Differenz exakt berechnen läßt.
	\end{itemize}
\end{exercisebox}

\begin{exercisebox}[Bestapproximation]
	Gegeben seien Messungen $x_1, x_2, \cdots, x_n\in\mathbb R$, $n\in \mathbb N$ einer Größe $x\in \mathbb R$. Gesucht ist eine Approximation $\tilde x\in \mathbb R$, deren Abstand zu allen Messwerten gleichzeitig minimal ist, also 
	\[ \sum\limits_{k=1}^n |x- x_k|^2 \stackrel{!}{=} \mathrm{min}\,.\]
		Bestimmen Sie eine Formel für $\tilde x$. 
\end{exercisebox} 

\begin{exercisebox}[Taylorentwicklung]
	Gegeben sei die Funktion $f: x\mapsto \cos(x)$. Bestimmen Sie die Taylorentwicklung $T_7$ von $f$ an der Stelle $x_0=0$ und schätzen Sie den Betrag des Lagrangeschen Restglieds $R_8$ im Intervall $x\in I= [ -\pi/6, \pi/6]$ ab. 
\end{exercisebox}

		\hline
		$y_i$ & $-3$ & $1$ & $0$ & $82$ \\

\end{center}

	\item[(a)]  Bestimmen Sie den Grad $n$ des Interpolationspolynoms. 
	\item[(b)]  Bestimmen Sie das lineare Gleichungssystem für die Koeffizienten des Interpolationspolynoms in der Darstellung \[ p(x) =a_nx^n+\cdots +a_1 x+a_0\] und lösen Sie es.

	%\item[(d)]  Bestimmen Sie das lineare Gleichungssystem für die Koeffizienten des Interpolationspolynoms in der Darstellung \[ q(x) =c_0+c_1(x-x_0) +c_2(x-x_0)(x-x_1) +\cdots +c_n(x-x_0)\cdots(x-x_{n-1})\] und lösen Sie es mit Matlab.
	%\item[(e)]  Berechnen Sie den Wert des Polynoms $q$ an der Zwischenstelle $\tilde x= 1$. Wie könnnen Sie den Wert effizient berechnen?
\end{itemize}
\end{exercisebox}


\begin{exercisebox}[Interpolation (Wiederholung)]
	Eine Funktion der Form \[ f: x \mapsto \frac{a_0}{2} + a_1 \cos(x) + b_1 \sin(x) \,,\] soll durch die Punkte $(x_i, y_i), i=1,2,3$ mit \[ (0|1), \, (\pi/6|-1),\, (\pi/2|-3)\] gelegt werden. Bestimmen Sie ein Gleichungssystem für die Koeffizienten $a_0, a_1, b_1$ und lösen Sie es.

\end{exercisebox}
